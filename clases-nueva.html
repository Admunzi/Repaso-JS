TUTORIAL JAVASCRIPT - Clases: Sintaxis Nueva

Constructor, set, get, privacidad, método, estático

constructor

Ejemplo:

class Restaurante {
    constructor(nombre, sillas, terraza) {
        this.nombre = nombre;
        this.sillas = sillas;
        this.terraza = terraza;
    }
}
let obj = new Restaurante('Rafalete', 20, true);

set

Ejemplo:

class Coche{
    set setColor(color){
        this.color = color;
    }
}

let obj = new Coche();
obj.color = "rojo";

get

Ejemplo:

class Aula{
    alumnos = 10;
    get getAlumnos(){
        return this.alumnos;
    }
}

let obj = new Aula();
console.log(obj.getAlumnos);

Privacidad

Para que una propiedad o metodo sea privado solo se tiene que añadir # antes

Ejemplo:

class ClaseConPropiedadPrivada {
    #propiedadPrivada;
}

class ClaseConMetodoPrivado {
    #metodoPrivado() {
        return 'hola mundo';
    }
}

class ClaseConPropiedadEstaticaPrivada {
    static #PROPIEDAD_ESTATICA_PRIVADA;
}

class ClaseConMetodoEstaticoPrivado {
    static #metodoEstaticoPrivado() {
        return 'hola mundo';
    }
}

Métodos

Ejemplo:

class Persona{
    saluda () {
        return "Hola mundo";
    }
}

let obj = new Persona();
console.log(obj.saluda());

static

Ejemplo:

class Persona{
    //Propiedad estática
    static genero = "Masculino";

    //Método estático
    static test() {
        console.log('Coche amarillo');
    }
}
